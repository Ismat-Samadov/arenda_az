  FROM DWH.D_CONTRACT_CREDIT D left join optimus.d_contract e on d.contract_ref_no =e.CONTRACT_REF_NO
 WHERE  
 d.BRANCH_CODE ='123'

SELECT D.EXPERT_NAME, count(*) say ,sum(round(d.initial_amount)) as amount
  FROM DWH.D_CONTRACT_CREDIT D
  left join optimus.d_contract e
    on d.contract_ref_no = e.CONTRACT_REF_NO
 WHERE --d.BRANCH_CODE = '123'
    e.COMPANY_NAME = n'AZƏRSU ASC-BAKI SUKANALI MMC'
 group by D.EXPERT_NAME
 order by say desc ;
