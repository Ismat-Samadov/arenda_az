SELECT D.START_DATE,
       d.contract_ref_no,
       D.PRODUCT_CODE,
       D.CMS_ID,
       ROUND(D.INITIAL_AMOUNT) AS AMOUNT,
       D.CUSTOMER_NAME,
       e.COMPANY_NAME,
       D.EXPERT_NAME
  FROM DWH.D_CONTRACT_CREDIT D
  left join optimus.d_contract e
    on d.contract_ref_no = e.CONTRACT_REF_NO
 WHERE e.cif='2632800'
 ORDER BY D.START_DATE;  --- d.cms_id = '4615491'
