SELECT C.PIN,C.FULL_NAME, COUNT(*) SAY
  FROM DWH.D_CONTRACT_CREDIT D INNER JOIN DWH.D_CUSTOMER C ON D.CIF =C.CIF
 WHERE START_DATE > TO_DATE('01.11.2022', 'DD.MM.YYYY')
   AND PRODUCT_CODE = 'ATKA'
 GROUP BY C.PIN,C.FULL_NAME
 ORDER BY SAY DESC;
