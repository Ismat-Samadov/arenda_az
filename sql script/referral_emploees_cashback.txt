SELECT /*+parallel */
    emp.fin,
    emp.name,
    emp.branch,
    emp.shobe,
    emp.bolme,
    emp.vezife,
    emp.vezife_2,
    SUM(a.cashback_amount) summa
FROM
         cms.f_cashback_transaction a
    INNER JOIN dwh.d_customer d ON a.cms_id = d.cms_id
    INNER JOIN fraud.emp_2    emp ON emp.fin = d.pin
WHERE
        termname = 'LOYALTY_REFERRAL_5'
    AND direction = 'CREDIT'
GROUP BY
    emp.fin,
    emp.name,
    emp.branch,
    emp.shobe,
    emp.bolme,
    emp.vezife,
    emp.vezife_2
ORDER BY
    summa DESC;
