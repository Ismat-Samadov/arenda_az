    round(AVG(round((sysdate - to_date(birth_date, 'dd.mm.yyyy')) / 360)))        AS mean_age,
    round(MEDIAN(round((sysdate - to_date(birth_date, 'dd.mm.yyyy')) / 360)))     AS median_age,
    round(STATS_MODE(round((sysdate - to_date(birth_date, 'dd.mm.yyyy')) / 360))) AS mod_age
FROM
    fraud.emp_2
