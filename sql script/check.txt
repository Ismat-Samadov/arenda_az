select a.bank_date,
       a.termlocation,
       a.termcountry,
       a.termcity,
       a.pan,
       a.org_amount,
       a.lcy_amount,
       a.org_currency
  from cms.f_transaction_details a
 where a.idclient = '12969792'
   and a.lcy_amount != a.org_currency
   and a.ORG_CURRENCY != 944
   and a.bank_date > to_date('01.04.2023', 'dd.mm.yyyy')
 order by bank_date;
