SELECT A.BANK_DATE,
       A.OP_DATE,
       A.CLIENT_ID,
       A.OP_STATUS,
       ROUND(A.AMOUNT) AMOUNT,
       ROUND(A.AMOUNT_AZN) AMOUNT_AZN,
       A.OP_NAME,
       A.CATEGORY_NAME,
       A.PROVIDER_NAME,
       A.IS_3DS
  FROM BIRBANK.F_OPERATION A
 WHERE A.BANK_DATE > TO_DATE('01.01.2022', 'DD.MM.YYYY')
   AND A.PROVIDER_NAME = n'Xalq Sığorta'
   AND A.CATEGORY_NAME = 'Insurance';
