create or replace view EMP_LOAN_PAYMENTS as
select   /* +parallel*/
         a.bank_date,
         a.credit_product, 
         a.amount_azn,
         d.PIN,
         d.NAME,
         d.SHOBE,
         d.VEZIFE
from birbank.f_operation a
inner join dwh.d_contract_credit c on (to_char(a.credit_product) = to_char(c.id) 
                                   or  to_char(a.credit_product) = to_char(c.contract_ref_no))
inner join fraud.emp_07_06_2023 d on a.client_id=d.USER_ID;
